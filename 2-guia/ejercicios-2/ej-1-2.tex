  \begin{enumerate}[label=\tt\alph*)]

    \item {\tt x $\oplus$ (y\por z) = (x $\oplus$ y)\por (x $\oplus$ z)} donde se aplica la propiedad distributiva con respecto a $\oplus$.
  \end{enumerate}
\end{enunciado}
